export function createAuth() {
  if (_authInstance) {
    return _authInstance;
  }

  try {
    const config = createAuthConfig();
    _authInstance = betterAuth(config);
    return _authInstance;
  } catch (error) {
    console.error('[createAuth] Failed to initialize auth:', error);
    throw error;
  }
}

// ✅ Working solution: Use ReturnType<typeof betterAuth> for proper typing
export const auth: ReturnType<typeof betterAuth> = new Proxy({} as any, {
  get(_, prop) {
    const authInstance = createAuth()
    return (authInstance as any)[prop]
  },
})

export type Auth = ReturnType<typeof createAuth>;
